## Install

```bash
npm install --save first-try
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'first-try'
import 'first-try/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
